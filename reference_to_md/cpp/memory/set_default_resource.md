[std::pmr::memory_resource](memory_resource.html)* set_default_resource( [std::pmr::memory_resource](memory_resource.html)* r ) noexcept; |  |  (since C++17)  
| |   
  
If r is not null, sets the default memory resource pointer to r; otherwise, sets the default memory resource pointer to [std::pmr::new_delete_resource](new_delete_resource.html)(). 

The _default memory resource pointer_ is used by certain facilities when an explicit memory resource is not supplied. The initial default memory resource pointer is the return value of [std::pmr::new_delete_resource](new_delete_resource.html "cpp/memory/new delete resource"). 

This function is thread-safe. Every call to `std::pmr::set_default_resource` _synchronizes with_ (see [std::memory_order](../atomic/memory_order.html "cpp/atomic/memory order")) the subsequent `std::pmr::set_default_resource` and [std::pmr::get_default_resource](get_default_resource.html "cpp/memory/get default resource") calls. 

### [[edit](https://en.cppreference.com/mwiki/index.php?title=cpp/memory/set_default_resource&action=edit&section=1 "Edit section: Return value")] Return value

Returns the previous value of the default memory resource pointer. 

### [[edit](https://en.cppreference.com/mwiki/index.php?title=cpp/memory/set_default_resource&action=edit&section=2 "Edit section: Example")] Example

Run this code
    
    
    #include <array>
    #include <cstddef>
    #include <cstdint>
    #include <iostream>
    #include <iterator>
    #include <memory_resource>
    #include <vector>
     
    class noisy_allocator : public [std::pmr::memory_resource](memory_resource.html)
    {
        void* do_allocate([std::size_t](../types/size_t.html) bytes, [std::size_t](../types/size_t.html) alignment) override
        {
            [std::cout](../io/cout.html) << "+ Allocating " << bytes << " bytes @ ";
            void* p = [std::pmr::new_delete_resource](new_delete_resource.html)()->allocate(bytes, alignment);
            [std::cout](../io/cout.html) << p << '\n';
            return p;
        }
     
        void do_deallocate(void* p, [std::size_t](../types/size_t.html) bytes, [std::size_t](../types/size_t.html) alignment) override
        {
            [std::cout](../io/cout.html) << "- Deallocating " << bytes << " bytes @ " << p << '\n';
            return [std::pmr::new_delete_resource](new_delete_resource.html)()->deallocate(p, bytes, alignment);
        }
     
        bool do_is_equal(const [std::pmr::memory_resource](memory_resource.html)& other) const noexcept override
        {
            return [std::pmr::new_delete_resource](new_delete_resource.html)()->is_equal(other);
        }
    };
     
    int main()
    {
        constexpr int push_back_limit{16};
        noisy_allocator mem;
        std::pmr::set_default_resource(&mem);
     
        {
            [std::cout](../io/cout.html) << "Entering scope #1 (without buffer on stack)...\n";
            [std::cout](../io/cout.html) << "Creating vector v...\n";
            [std::pmr::vector](../container/vector.html)<[std::uint16_t](../types/integer.html)> v{1, 2, 3, 4};
            [std::cout](../io/cout.html) << "v.data() @ " << v.data() << '\n';
     
            [std::cout](../io/cout.html) << "Requesting more...\n";
            for (int i{0}; i != push_back_limit; ++i)
            {
                v.push_back(i);
                [std::cout](../io/cout.html) << "v.size(): " << v.size() << '\n';
            }
            [std::cout](../io/cout.html) << "Exiting scope #1...\n";
        }
     
        [std::cout](../io/cout.html) << '\n';
     
        {
            [std::cout](../io/cout.html) << "Entering scope #2 (with buffer on stack)...\n";
     
            [std::uint8_t](../types/integer.html) buffer[16];
            [std::cout](../io/cout.html) << "Allocating buffer on stack: " << sizeof buffer << " bytes @ "
                      << static_cast<void*>(buffer) << '\n';
            [std::pmr::monotonic_buffer_resource](monotonic_buffer_resource.html) mem_res{[std::data](../iterator/data.html)(buffer), [std::size](../iterator/size.html)(buffer)};
     
            [std::cout](../io/cout.html) << "Creating vector v...\n";
            [std::pmr::vector](../container/vector.html)<[std::uint16_t](../types/integer.html)> v{{1, 2, 3, 4}, &mem_res};
            [std::cout](../io/cout.html) << "v.data() @ " << v.data() << '\n'; // equals to `buffer` address
     
            [std::cout](../io/cout.html) << "Requesting more...\n";
            for (int i{0}; i != push_back_limit; ++i)
            {
                v.push_back(i);
                [std::cout](../io/cout.html) << "v.size(): " << v.size() << '\n';
            }
            [std::cout](../io/cout.html) << "Exiting scope #2...\n";
        }
    }
