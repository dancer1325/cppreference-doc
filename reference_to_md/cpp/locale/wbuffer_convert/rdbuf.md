1) Returns the pointer to the underlying byte stream.

2) Replaces the associated byte stream with bytebuf.

### Return value

1) `_[bufptr](../wbuffer_convert.html#bufptr "cpp/locale/wbuffer convert")_`

2) the previous value of `_[bufptr](../wbuffer_convert.html#bufptr "cpp/locale/wbuffer convert")_`

### Example

Run this code
    
    
    #include <codecvt>
    #include <iostream>
    #include <locale>
    #include <sstream>
    ¬†
    int main()
    {
        // Convert UTF-8 to UCS4
        [std::stringbuf](../../io/basic_stringbuf.html) utf8buf("z\u00df\u6c34\U0001d10b");
                           // or "\x7a\xc3\x9f\xe6\xb0\xb4\xf0\x9d\x84\x8b"
                           // or u8"z√üÊ∞¥ùÑã"
        [std::wbuffer_convert](../wbuffer_convert.html)<[std::codecvt_utf8](../codecvt_utf8.html)<wchar_t>> conv(&utf8buf);
        [std::wistream](../../io/basic_istream.html) ucsbuf(&conv);
        [std::cout](../../io/cout.html) << "Reading from a UTF-8 stringbuf via wbuffer_convert: "
                  << [std::hex](../../io/manip/hex.html) << [std::showbase](../../io/manip/showbase.html);
        for (wchar_t c; ucsbuf.get(c);)
            [std::cout](../../io/cout.html) << static_cast<[std::wint_t](../../string/wide.html)>(c) << ' ';
    ¬†
        // Reuse the same wbuffer_convert to handle UCS4 to UTF-8 output
        conv.rdbuf([std::cout](../../io/cout.html).rdbuf());
        [std::wostream](../../io/basic_ostream.html) out(&conv);
        [std::cout](../../io/cout.html) << "\nSending UCS4 data to std::cout via wbuffer_convert: ";
        out << L"z\u00df\u6c34\U0001d10b\n";
    }

Output:
