### Parameters

v  |  \-  |  the elements to clamp   
---|---|---  
lo, hi  |  \-  |  the boundaries to clamp v to   
  
### Return value

The result of element-wise application of [std::clamp](../../algorithm/clamp.html)(v[i], lo[i], hi[i]) for all i ∈ `[`​0​`, `size()`)`. 

### Example

Run this code
    
    
    #include <cstddef>
    #include <cstdint>
    #include <experimental/simd>
    #include <iomanip>
    #include <iostream>
    namespace stdx = std::experimental;
     
    void println(auto rem, auto const v)
    {
        [std::cout](../../io/cout.html) << rem << ": ";
        for ([std::size_t](../../types/size_t.html) i = 0; i != v.size(); ++i)
            [std::cout](../../io/cout.html) << [std::setw](../../io/manip/setw.html)(4) << v[i] << ' ';
        [std::cout](../../io/cout.html) << '\n';
    }
     
    int main()
    {
        stdx::fixed_size_simd<int, 8> a{[](int i) {
            static constexpr auto c = {-129, -128, -1, 0, 42, 127, 128, 255};
            return c.begin()[i];
        }};
        println("a", a);
     
        stdx::fixed_size_simd<int, 8> lo1{[INT8_MIN](../../types/integer.html)};
        stdx::fixed_size_simd<int, 8> hi1{[INT8_MAX](../../types/integer.html)};
        const auto b = stdx::clamp(a, lo1, hi1);
        println("b", b);
     
        stdx::fixed_size_simd<int, 8> lo2{0};
        stdx::fixed_size_simd<int, 8> hi2{[UINT8_MAX](../../types/integer.html)};
        const auto c = stdx::clamp(a, lo2, hi2);
        println("c", c);
    }

Output: 
    
    
    a: -129 -128   -1    0   42  127  128  255 
    b: -128 -128   -1    0   42  127  127  127 
    c:    0    0    0    0   42  127  128  255
