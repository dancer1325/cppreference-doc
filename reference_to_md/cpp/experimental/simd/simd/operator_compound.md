Applies the given compound assignment operator element-wise to each corresponding element of the operands such that for all i in the range of `[`​0​`, `[`size()`](size.html "cpp/experimental/simd/simd/size")`)` the result is equivalent to: 

1) lhs[i] += rhs[i]

2) lhs[i] -= rhs[i]

3) lhs[i] *= rhs[i]

4) lhs[i] /= rhs[i]

5) lhs[i] %= rhs[i]

6) lhs[i] &= rhs[i]

7) lhs[i] |= rhs[i]

8) lhs[i] ^= rhs[i]

9) lhs[i] <<= rhs[i]

10) lhs[i] <<= n

11) lhs[i] >>= rhs[i]

12) lhs[i] >>= n

### [[edit](https://en.cppreference.com/mwiki/index.php?title=cpp/experimental/simd/simd/operator_compound&action=edit&section=3 "Edit section: Example")] Example

Computes cosh(β)² - sinh(β)² (== 1) for some arbitrary angle β, on SIMD data.

Run this code
    
    
    #include <cstddef>
    #include <experimental/simd>
    #include <iomanip>
    #include <iostream>
    #include <string_view>
    namespace stq = std::experimental;
     
    void print(const [std::string_view](../../../string/basic_string_view.html) rem, const auto x)
    {
        [std::cout](../../../io/cout.html) << rem << '\n' << [std::fixed](../../../io/manip/fixed.html);
        for ([std::size_t](../../../types/size_t.html) t{}; t != x.size(); ++t)
            [std::cout](../../../io/cout.html) << [std::setw](../../../io/manip/setw.html)(10) << x[t] << ((t + 1) % 8 ? ',' : '\n');
        [std::cout](../../../io/cout.html) << '\n';
    }
     
    int main()
    {
        constexpr auto N = 32UZ;
        using elem_t = long double;
     
        stq::fixed_size_simd<elem_t, N> x{[](elem_t i) { return i / N; }};
        stq::fixed_size_simd<elem_t, N> y{x};
        x = stq::cosh(x);
        y = stq::sinh(y);
        x *= x;
        y *= y;
        print("cosh(x)²:", x);
        print("sinh(x)²:", y);
        auto z = x - y;
        print("cosh(x)² - sinh(x)²:", z);
        [std::cout](../../../io/cout.html) << "reduce(z) = " << stq::reduce(z) << '\n';
    }
