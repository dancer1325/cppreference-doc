wchar_t* fgetws( wchar_t* str, int count, [std::FILE](FILE.html)* stream ); |  |   
| |   
  
Reads at most count - 1 wide characters from the given file stream and stores them in str. The produced wide string is always null-terminated. Parsing stops if end-of-file occurs or a newline wide character is found, in which case str will contain that wide newline character.

str  |  \-  |  wide string to read the characters to   
---|---|---  
count  |  \-  |  the length of str  
stream  |  \-  |  file stream to read the data from   
  
### [[edit](https://en.cppreference.com/mwiki/index.php?title=cpp/io/c/fgetws&action=edit&section=2 "Edit section: Return value")] Return value

### [[edit](https://en.cppreference.com/mwiki/index.php?title=cpp/io/c/fgetws&action=edit&section=3 "Edit section: Example")] Example

Run this code
    
    
    #include <array>
    #include <clocale>
    #include <cstdio>
    #include <cstdlib>
    #include <cwchar>
    #include <cwctype>
    #include <iomanip>
    #include <iostream>
    #include <span>
    #include <string>
     
    void dump([std::span](../../container/span.html)<const wchar_t> sp, [std::size_t](../../types/size_t.html) width = 14)
    {
        for (wchar_t wc : sp)
            [std::wcout](../cout.html) << ([std::iswprint](../../string/wide/iswprint.html)(wc) ? wc : L'.');
        [std::wcout](../cout.html) << [std::wstring](../../string/basic_string.html)(width > sp.size() ? width - sp.size() : 1, L' ')
                   << [std::hex](../manip/hex.html) << [std::uppercase](../manip/uppercase.html) << [std::setfill](../manip/setfill.html)(L'0');
        for (wchar_t wc : sp)
            [std::wcout](../cout.html) << [std::setw](../manip/setw.html)(sizeof wc) << static_cast<unsigned>(wc) << ' ';
        [std::wcout](../cout.html) << '\n';
    }
     
    int main()
    {
        // Create temp file that contains wide characters
        [std::setlocale](../../locale/setlocale.html)([LC_ALL](../../locale/LC_categories.html), "en_US.utf8");
        [std::FILE](FILE.html)* tmpf = [std::tmpfile](tmpfile.html)();
     
        for (const wchar_t* text : {
            L"Tétraèdre"    L"\n",
            L"Cube"         L"\n",
            L"Octaèdre"     L"\n",
            L"Icosaèdre"    L"\n",
            L"Dodécaèdre"   L"\n"
            })
            if (int rc = [std::fputws](fputws.html)(text, tmpf); rc == [EOF](../c.html))
            {
                [std::perror](perror.html)("fputws()"); // POSIX requires that errno is set
                return [EXIT_FAILURE](../../utility/program/EXIT_status.html);
            }
     
        [std::rewind](rewind.html)(tmpf);
     
        [std::array](../../container/array.html)<wchar_t, 12> buf;
        while (std::fgetws(buf.data(), buf.size(), tmpf) != nullptr)
            dump([std::span](../../container/span.html)(buf.data(), buf.size()));
     
        return [EXIT_SUCCESS](../../utility/program/EXIT_status.html);
    }
