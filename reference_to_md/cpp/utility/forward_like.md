template< class T, class U >  
constexpr auto&& forward_like( U&& x ) noexcept; |  |  (since C++23)  
| |   
  
Returns a reference to x which has similar properties to `T&&`. 

The return type is determined as below: 

  1. If [std::remove_reference_t](../types/remove_reference.html)<T> is a const-qualified type, then the referenced type of the return type is const [std::remove_reference_t](../types/remove_reference.html)<U>. Otherwise, the referenced type is [std::remove_reference_t](../types/remove_reference.html)<U>. 
  2. If `T&&` is an lvalue reference type, then the return type is also an lvalue reference type. Otherwise, the return type is an rvalue reference type. 



If `T` is not a [referenceable type](../meta.html#Definitions "cpp/meta"), the program is ill-formed.

### Parameters

x  |  \-  |  a value needs to be forwarded like type `T`  
---|---|---  
  
### Return value

A reference to x of the type determined as above. 

### Notes

Like [std::forward](forward.html "cpp/utility/forward"), [`std::move`](move.html "cpp/utility/move"), and [std::as_const](as_const.html "cpp/utility/as const"), `std::forward_like` is a type cast that only influences the [value category](../language/value_category.html "cpp/language/value category") of an expression, or potentially adds const-qualification. 

When `m` is an actual member and thus o.m a valid expression, this is usually spelled as [std::forward](forward.html)<decltype(o)>(o).m in C++20 code. 

This leads to three possible models, called _merge_ , _tuple_ , and _language_. 

  * _merge_ : merge the const qualifiers, and adopt the value category of the `Owner`. 
  * _tuple_ : what std::get<0>(Owner) does, assuming `Owner` is a [std::tuple](tuple.html)<Member>. 
  * _language_ : what [std::forward](forward.html)<decltype(Owner)>(o).m does. 



The main scenario that `std::forward_like` caters to is adapting “far” objects. Neither the _tuple_ nor the _language_ scenarios do the right thing for that main use-case, so the _merge_ model is used for `std::forward_like`.

### Possible implementation
    
    
    template<class T, class U>
    constexpr auto&& forward_like(U&& x) noexcept
    {
        constexpr bool is_adding_const = [std::is_const_v](../types/is_const.html)<[std::remove_reference_t](../types/remove_reference.html)<T>>;
        if constexpr ([std::is_lvalue_reference_v](../types/is_lvalue_reference.html)<T&&>)
        {
            if constexpr (is_adding_const)
                return [std::as_const](as_const.html)(x);
            else
                return static_cast<U&>(x);
        }
        else
        {
            if constexpr (is_adding_const)
                return std::move([std::as_const](as_const.html)(x));
            else
                return std::move(x);
        }
    }

### Example

Run this code
    
    
    #include <cstddef>
    #include <iostream>
    #include <memory>
    #include <optional>
    #include <type_traits>
    #include <utility>
    #include <vector>
     
    struct TypeTeller
    {
        void operator()(this auto&& self)
        {
            using SelfType = decltype(self);
            using UnrefSelfType = [std::remove_reference_t](../types/remove_reference.html)<SelfType>;
            if constexpr ([std::is_lvalue_reference_v](../types/is_lvalue_reference.html)<SelfType>)
            {
                if constexpr ([std::is_const_v](../types/is_const.html)<UnrefSelfType>)
                    [std::cout](../io/cout.html) << "const lvalue\n";
                else
                    [std::cout](../io/cout.html) << "mutable lvalue\n";
            }
            else
            {
                if constexpr ([std::is_const_v](../types/is_const.html)<UnrefSelfType>)
                    [std::cout](../io/cout.html) << "const rvalue\n";
                else
                    [std::cout](../io/cout.html) << "mutable rvalue\n";
            }
        }
    };
     
    struct FarStates
    {
        [std::unique_ptr](../memory/unique_ptr.html)<TypeTeller> ptr;
        [std::optional](optional.html)<TypeTeller> opt;
        [std::vector](../container/vector.html)<TypeTeller> container;
     
        auto&& from_opt(this auto&& self)
        {
            return std::forward_like<decltype(self)>(self.opt.value());
            // It is OK to use std::forward<decltype(self)>(self).opt.value(),
            // because std::optional provides suitable accessors.
        }
     
        auto&& operator[](this auto&& self, [std::size_t](../types/size_t.html) i)
        {
            return std::forward_like<decltype(self)>(self.container.at(i));
            // It is not so good to use std::forward<decltype(self)>(self)[i], because
            // containers do not provide rvalue subscript access, although they could.
        }
     
        auto&& from_ptr(this auto&& self)
        {
            if (!self.ptr)
                throw [std::bad_optional_access](bad_optional_access.html){};
            return std::forward_like<decltype(self)>(*self.ptr);
            // It is not good to use *std::forward<decltype(self)>(self).ptr, because
            // std::unique_ptr<TypeTeller> always dereferences to a non-const lvalue.
        }
    };
     
    int main()
    {
        FarStates my_state
        {
            .ptr{[std::make_unique](../memory/unique_ptr/make_unique.html)<TypeTeller>()},
            .opt{[std::in_place](in_place.html), TypeTeller{}},
            .container{[std::vector](../container/vector.html)<TypeTeller>(1)},
        };
     
        my_state.from_ptr()();
        my_state.from_opt()();
        my_state[0]();
     
        [std::cout](../io/cout.html) << '\n';
     
        [std::as_const](as_const.html)(my_state).from_ptr()();
        [std::as_const](as_const.html)(my_state).from_opt()();
        [std::as_const](as_const.html)(my_state)[0]();
     
        [std::cout](../io/cout.html) << '\n';
     
        std::move(my_state).from_ptr()();
        std::move(my_state).from_opt()();
        std::move(my_state)[0]();
     
        [std::cout](../io/cout.html) << '\n';
     
        std::move([std::as_const](as_const.html)(my_state)).from_ptr()();
        std::move([std::as_const](as_const.html)(my_state)).from_opt()();
        std::move([std::as_const](as_const.html)(my_state))[0]();
     
        [std::cout](../io/cout.html) << '\n';
    }
