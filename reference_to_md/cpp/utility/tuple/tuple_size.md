### Example

Run this code
    
    
    #include <iostream>
    #include <tuple>
     
    template <class T>
    void test(T value)
    {
        int a[[std::tuple_size_v](../tuple_size.html)<T>]; // can be used at compile time
     
        [std::cout](../../io/cout.html) << [std::tuple_size](../tuple_size.html)<T>{} << ' ' // or at run time
                  << sizeof a << ' '
                  << sizeof value << '\n';
    }
     
    int main()
    {
        test([std::make_tuple](make_tuple.html)(1, 2, 3.14));
    }

Possible output: 
    
    
    3 12 16
