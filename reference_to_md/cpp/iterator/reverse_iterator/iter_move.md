friend constexpr [std::iter_rvalue_reference_t](../iter_t.html)<Iter>  
iter_move( const [std::reverse_iterator](../reverse_iterator.html)& i ) noexcept(/* see below */); |  |  (since C++20)  
---|---|---  
| |   
  
Casts the result of dereferencing the adjusted underlying iterator to its associated rvalue reference type. 

Equivalent to auto tmp = i.base();  
return std::[ranges::iter_move](../ranges/iter_move.html)(\--tmp);. 

This function template is not visible to ordinary [unqualified](../../language/unqualified_lookup.html "cpp/language/unqualified lookup") or [qualified lookup](../../language/qualified_lookup.html "cpp/language/qualified lookup"), and can only be found by [argument-dependent lookup](../../language/adl.html "cpp/language/adl") when [std::reverse_iterator](../reverse_iterator.html)<Iter> is an associated class of the arguments.

### Example

Run this code
    
    
    #include <iomanip>
    #include <iostream>
    #include <iterator>
    #include <string>
    #include <vector>
     
    void print(const auto& rem, const auto& v)
    {
        [std::cout](../../io/cout.html) << rem << '[' << size(v) << "] {";
        for (char comma[]{0, 0}; const auto& s : v)
            [std::cout](../../io/cout.html) << comma << ' ' << [std::quoted](../../io/manip/quoted.html)(s), comma[0] = ',';
        [std::cout](../../io/cout.html) << " }\n";
    }
     
    int main()
    {
        [std::vector](../../container/vector.html)<[std::string](../../string/basic_string.html)> p{"Alpha", "Bravo", "Charlie"}, q;
     
        print("p", p), print("q", q);
     
        using RI = [std::reverse_iterator](../reverse_iterator.html)<[std::vector](../../container/vector.html)<[std::string](../../string/basic_string.html)>::iterator>;
     
        for (RI iter{p.rbegin()}, rend{p.rend()}; iter != rend; ++iter)
            q.emplace_back(/* ADL */ iter_move(iter));
     
        print("p", p), print("q", q);
    }

Possible output: 
    
    
    p[3] { "Alpha", "Bravo", "Charlie" }
    q[0] { }
    p[3] { "", "", "" }
    q[3] { "Charlie", "Bravo", "Alpha" }
