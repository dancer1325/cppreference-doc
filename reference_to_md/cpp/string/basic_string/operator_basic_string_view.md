operator [std::basic_string_view](../basic_string_view.html)<CharT, Traits>() const noexcept; |  |  (since C++17)   
(constexpr since C++20)

| |   
  
Returns a [std::basic_string_view](../basic_string_view.html "cpp/string/basic string view"), constructed as if by [std::basic_string_view](../basic_string_view.html)<CharT, Traits>(data(), size()).

---  
  
### Parameters

(none) 

### Return value

A string view representing the entire contents of the string. 

### Notes

It is the programmer's responsibility to ensure that the resulting string view does not outlive the string. 
    
    
    [std::string](../basic_string.html) get_string();
    int f([std::string_view](../basic_string_view.html) sv);
     
    int x = f(get_string()); // OK
    [std::string_view](../basic_string_view.html) sv = get_string(); // Bad: holds a dangling pointer

### Example

Run this code
    
    
    #include <iostream>
    #include <string>
    #include <string_view>
     
    void show_wstring_size([std::wstring_view](../basic_string_view.html) wcstr_v)
    {
        [std::cout](../../io/cout.html) << wcstr_v.size() << " code points\n";
    }
     
    int main()
    {
        [std::string](../basic_string.html) cppstr = "ラーメン";   // narrow string
        [std::wstring](../basic_string.html) wcstr = L"ラーメン";  // wide string
     
        // Implicit conversion from string to string_view
        // via std::string::operator string_view:
        [std::string_view](../basic_string_view.html) cppstr_v = cppstr;
     
        [std::cout](../../io/cout.html) << cppstr_v << '\n'
                  << cppstr_v.size() << " code units\n";
     
        // Implicit conversion from wstring to wstring_view
        // via std::wstring::operator wstring_view:
        show_wstring_size(wcstr);
    }

Output: 
    
    
    ラーメン
    12 code units
    4 code points
