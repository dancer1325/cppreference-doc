size_type max_size() const; |  | (noexcept since C++11)   
(constexpr since C++20)  
---|---|---  
| |   
  
Returns the maximum number of elements the string is able to hold due to system or library implementation limitations, i.e. [std::distance](../../iterator/distance.html)(begin(), end()) for the largest string.

### [[edit](https://en.cppreference.com/mwiki/index.php?title=cpp/string/basic_string/max_size&action=edit&section=4 "Edit section: Example")] Example

Run this code
    
    
    #include <array>
    #include <climits>
    #include <iomanip>
    #include <iostream>
    #include <locale>
    #include <string>
    #include <typeinfo>
     
    #include <boost/core/demangle.hpp>
     
    template<typename T>
    void print_basic_string_max_size()
    {
        [std::basic_string](../basic_string.html)<T> s;
        auto max_size = s.max_size();
        [std::cout](../../io/cout.html).imbue([std::locale](../../locale/locale.html)("en_US.UTF-8"));
        [std::cout](../../io/cout.html) << "basic_string<" << boost::core::demangle(typeid(T).name())
                  << ">:\n\t" << max_size << " = ";
        [std::cout](../../io/cout.html) << [std::setprecision](../../io/manip/setprecision.html)(3) << (double) max_size << " = ";
        [std::cout](../../io/cout.html).imbue([std::locale](../../locale/locale.html)("C"));
        [std::cout](../../io/cout.html) << [std::hex](../../io/manip/hex.html) << [std::setfill](../../io/manip/setfill.html)('0') << "0x"
                  << [std::setw](../../io/manip/setw.html)(sizeof(typename decltype(s)::size_type) << 1)
                  << max_size << '\n' << [std::dec](../../io/manip/hex.html);
    };
     
    int main()
    {
        [std::cout](../../io/cout.html) << "Pointer size: " << [CHAR_BIT](../../types/climits.html) * sizeof(void*) << " bits\n"
                     "Maximum sizes:\n";
     
        print_basic_string_max_size<char>();
        print_basic_string_max_size<char16_t>();
        print_basic_string_max_size<char32_t>();
        print_basic_string_max_size<wchar_t>();
        print_basic_string_max_size<long>();
     
        using CharT = [std::array](../../container/array.html)<char, 01232>;
        print_basic_string_max_size<CharT>();
    }
