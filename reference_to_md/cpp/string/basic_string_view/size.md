### [[edit](https://en.cppreference.com/mwiki/index.php?title=cpp/string/basic_string_view/size&action=edit&section=4 "Edit section: Example")] Example

Run this code
    
    
    #include <iostream>
    #include <string_view>
     
    // Print a string surrounded by single quotes, its
    // length and whether it is considered empty.
    void check_string([std::string_view](../basic_string_view.html) ref)
    {
        [std::cout](../../io/cout.html) << [std::boolalpha](../../io/manip/boolalpha.html)
                  << "'" << ref << "' has " << ref.size()
                  << " character(s); emptiness: " << ref.empty() << '\n';
    }
     
    int main(int argc, char **argv)
    {
        // An empty string
        check_string("");
     
        // Almost always not empty: argv[0]
        if (argc > 0)
            check_string(argv[0]);
    }

Possible output: 
    
    
    '' has 0 character(s); emptiness: true
    './a.out' has 7 character(s); emptiness: false
