`std::gslice_array` is a helper template used by the [valarray subscript operator](operator_at.html "cpp/numeric/valarray/operator at") with [std::gslice](gslice.html) argument. It has reference semantics to a subset of the array specified by the [std::gslice](gslice.html) object.

### Example

Run this code
    
    
    #include <cassert>
    #include <cstddef>
    #include <iomanip>
    #include <iostream>
    #include <numeric>
    #include <valarray>
     
    int main()
    {
        [std::valarray](../valarray.html)<int> data(32);
        [std::iota](../../algorithm/iota.html)([std::begin](../../iterator/begin.html)(data), [std::end](../../iterator/end.html)(data), 0);
     
        const [std::size_t](../../types/size_t.html) offset = 1, z = 2, y = 3, x = 4;
        const [std::valarray](../valarray.html)<[std::size_t](../../types/size_t.html)> sizes{z, y, x};
        const [std::valarray](../valarray.html)<[std::size_t](../../types/size_t.html)> strides{15, 5, 1};
        const [std::gslice](gslice.html) gslice = [std::gslice](gslice.html)(offset, sizes, strides);
        // Indices are generated according to the formula:
        // index[k] = offset + [0,1,2)*15 + [0,1,2,3)*5 + [0,1,2,3,4)*1
        //          = offset + inner_product(sizes[k], strides);
        // where sizes[k] = {[0,z), [0,y), [0,x)}, while the rightmost index (x)
        // runs fastest. As a result we have following set of indices:
        //  index[0]  = 1 + 0*15 + 0*5 + 0*1 = 1
        //  index[1]  = 1 + 0*15 + 0*5 + 1*1 = 2
        //  index[2]  = 1 + 0*15 + 0*5 + 2*1 = 3
        //  index[3]  = 1 + 0*15 + 0*5 + 3*1 = 4
        //  index[4]  = 1 + 0*15 + 1*5 + 0*1 = 6
        //  index[5]  = 1 + 0*15 + 1*5 + 1*1 = 7
        //  index[6]  = 1 + 0*15 + 1*5 + 2*1 = 8
        //  index[7]  = 1 + 0*15 + 1*5 + 3*1 = 9
        //  ...
        //  index[22] = 1 + 1*15 + 2*5 + 2*1 = 28
        //  index[23] = 1 + 1*15 + 2*5 + 3*1 = 29
     
        const [std::valarray](../valarray.html)<int> indices = data[gslice];
        for (unsigned i = 0; i != indices.size(); ++i)
            [std::cout](../../io/cout.html) << [std::setfill](../../io/manip/setfill.html)('0') << [std::setw](../../io/manip/setw.html)(2) << indices[i] << ' ';
        [std::cout](../../io/cout.html) << "\nTotal indices: " << indices.size() << '\n';
        [assert](../../error/assert.html)(indices.size() == x * y * z);
     
        data = 0;
        std::gslice_array<int> gslice_array = data[gslice];
        gslice_array = 1;
        // Cells that correspond to generated indices = '1', skipped cells = '0'.
        for (auto i : data)
            [std::cout](../../io/cout.html) << i << ' ';
        [std::cout](../../io/cout.html) << "\nSum of ones = " << data.sum() << '\n';
    }

    0 1 1 1 1 0 1 1 1 1 0 1 1 1 1 0 1 1 1 1 0 1 1 1 1 0 1 1 1 1 0 0 
    Sum of ones = 24
