Returns the maximum finite value representable by the numeric type `T`. Meaningful for all bounded types. 

### Return value

### Example

Demonstrates the use of `max()` with some fundamental types and some standard library typedefs (the output is system-specific):

Run this code
    
    
    #include <boost/type_index.hpp>
    #include <cstddef>
    #include <iomanip>
    #include <iostream>
    #include <limits>
    #include <type_traits>
     
    template<typename T>
    void print_max_value_of()
    {
        constexpr T max{[std::numeric_limits](../numeric_limits.html)<T>::max()};
     
        [std::cout](../../io/cout.html) << [std::setw](../../io/manip/setw.html)(16) << boost::typeindex::type_id<T>() << ": ";
        if constexpr ([std::is_floating_point_v](../is_floating_point.html)<T>)
            [std::cout](../../io/cout.html) << [std::defaultfloat](../../io/manip/fixed.html) << max << " = " << [std::hexfloat](../../io/manip/fixed.html) << max << '\n';
        else
        {
            constexpr auto m{static_cast<unsigned long long>(max)};
            [std::cout](../../io/cout.html) << [std::dec](../../io/manip/hex.html) << m << " = " << [std::hex](../../io/manip/hex.html) << m << '\n';
        }
    }
     
    int main()
    {
        [std::cout](../../io/cout.html) << [std::showbase](../../io/manip/showbase.html);
     
        print_max_value_of<bool>();
        print_max_value_of<short>();
        print_max_value_of<int>();
        print_max_value_of<[std::streamsize](../../io/streamsize.html)>();
        print_max_value_of<[std::size_t](../size_t.html)>();
        print_max_value_of<char>();
        print_max_value_of<char16_t>();
        print_max_value_of<wchar_t>();
        print_max_value_of<float>();
        print_max_value_of<double>();
        print_max_value_of<long double>();
    }
