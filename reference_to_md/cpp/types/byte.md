enum class byte : unsigned char {}; |  |  (since C++17)  
| |   
  
`std::byte` is a distinct type that implements the concept of byte as specified in the C++ language definition. 

Like unsigned char, it can be used to access raw memory occupied by other objects ([object representation](../language/objects.html "cpp/language/object")), but unlike unsigned char, it is not a character type and is not an arithmetic type. `std::byte` models a mere collection of bits, supporting only bitshift operations with an integer, and bitwise and comparison operations with another `std::byte`.

### Notes

A numeric value n can be converted to a byte value using std::byte{n}, due to C++17 [relaxed enum class initialization](../language/enum.html#enum_relaxed_init_cpp17 "cpp/language/enum") rules. 

A byte can be converted to a numeric value (such as to produce an integer hash of an object) the usual way with an [explicit conversion](../language/explicit_cast.html "cpp/language/explicit cast") or alternatively with [`std::to_integer`](byte.html#Non-member_functions).

### Example

Run this code
    
    
    #include <bitset>
    #include <cassert>
    #include <cstddef>
    #include <iostream>
    #include <utility>
     
    [std::ostream](../io/basic_ostream.html)& operator<<([std::ostream](../io/basic_ostream.html)& os, std::byte b)
    {
        return os << [std::bitset](../utility/bitset.html)<8>(std::to_integer<int>(b));
    }
     
    int main()
    {
        // std::byte y = 1; // Error: cannot convert int to byte.
        std::byte y{1}; // OK
     
        // if (y == 13) {} // Error: cannot be compared.
        if (y == std::byte{13}) {} // OK, bytes are comparable
     
        int arr[]{1, 2, 3};
        // int c = a[y]; // Error: array subscript is not an integer
        [[maybe_unused]] int i = arr[std::to_integer<int>(y)]; // OK
        [[maybe_unused]] int j = arr[[std::to_underlying](../utility/to_underlying.html)(y)];   // OK
     
        auto to_int = [](std::byte b) { return std::to_integer<int>(b); };
     
        std::byte b{42};
        [assert](../error/assert.html)(to_int(b) == 0b00101010);
        [std::cout](../io/cout.html) << b << '\n';
     
        // b *= 2; // Error: b is not of arithmetic type
        b <<= 1;
        [assert](../error/assert.html)(to_int(b) == 0b01010100);
     
        b >>= 1;
        [assert](../error/assert.html)(to_int(b) == 0b00101010);
     
        [assert](../error/assert.html)(to_int(b << 1) == 0b01010100);
        [assert](../error/assert.html)(to_int(b >> 1) == 0b00010101);
     
        b |= std::byte{0b11110000};
        [assert](../error/assert.html)(to_int(b) == 0b11111010);
     
        b &= std::byte{0b11110000};
        [assert](../error/assert.html)(to_int(b) == 0b11110000);
     
        b ^= std::byte{0b11111111};
        [assert](../error/assert.html)(to_int(b) == 0b00001111);
    }

Output: 
    
    
    00101010
