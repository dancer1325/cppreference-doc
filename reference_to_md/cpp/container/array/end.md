Returns an iterator past the last element of *this. 

This returned iterator only acts as a sentinel. It is not guaranteed to be [dereferenceable](../../iterator.html#Dereferenceability_and_validity "cpp/iterator").

### [[edit](https://en.cppreference.com/mwiki/index.php?title=Template:cpp/container/end&action=edit&section=T-4 "Template:cpp/container/end")] Example

Run this code
    
    
    #include <algorithm>
    #include <array>
    #include <iomanip>
    #include <iostream>
     
    int main()
    {
        [std::cout](../../io/cout.html) << [std::boolalpha](../../io/manip/boolalpha.html);
     
        [std::array](../array.html)<int, 0> empty;
        [std::cout](../../io/cout.html) << "1) "
                  << (empty.begin() == empty.end()) << ' '     // true
                  << (empty.cbegin() == empty.cend()) << '\n'; // true
        // *(empty.begin()) = 42; // => undefined behavior at run-time
     
     
        [std::array](../array.html)<int, 4> numbers{5, 2, 3, 4};
        [std::cout](../../io/cout.html) << "2) "
                  << (numbers.begin() == numbers.end()) << ' '    // false
                  << (numbers.cbegin() == numbers.cend()) << '\n' // false
                  << "3) "
                  << *(numbers.begin()) << ' '    // 5
                  << *(numbers.cbegin()) << '\n'; // 5
     
        *numbers.begin() = 1;
        [std::cout](../../io/cout.html) << "4) " << *(numbers.begin()) << '\n'; // 1
        // *(numbers.cbegin()) = 42; // compile-time error: 
                                     // read-only variable is not assignable
     
        // print out all elements
        [std::cout](../../io/cout.html) << "5) ";
        [std::for_each](../../algorithm/for_each.html)(numbers.cbegin(), numbers.cend(), [](int x)
        {
            [std::cout](../../io/cout.html) << x << ' ';
        });
        [std::cout](../../io/cout.html) << '\n';
     
        constexpr [std::array](../array.html) constants{'A', 'B', 'C'};
        static_assert(constants.begin() != constants.end());   // OK
        static_assert(constants.cbegin() != constants.cend()); // OK
        static_assert(*constants.begin() == 'A');              // OK
        static_assert(*constants.cbegin() == 'A');             // OK
        // *constants.begin() = 'Z'; // compile-time error: 
                                     // read-only variable is not assignable
    }

Output: 
    
    
    1) true true
    2) false false
    3) 5 5
    4) 1
    5) 1 2 3 4
