constexpr iterator begin() const noexcept; |  (1)  |  (since C++20)  
---|---|---  
constexpr const_iterator cbegin() const noexcept; |  (2)  |  (since C++23)  
| |   
  
Returns an iterator to the first element of *this. 

If *this is empty, the returned iterator will be equal to end().

### [[edit](https://en.cppreference.com/mwiki/index.php?title=Template:cpp/container/begin&action=edit&section=T-4 "Template:cpp/container/begin")] Example

Run this code
    
    
    #include <iostream>
    #include <span>
     
    void print([std::span](../span.html)<const int> array)
    {
        [std::cout](../../io/cout.html) << "array = ";
        for (auto it = array.begin(); it != array.end(); ++it)
            [std::cout](../../io/cout.html) << *it << ' ';
        [std::cout](../../io/cout.html) << '\n';
    }
     
    void set_first_element([std::span](../span.html)<int> sp, int new_value)
    {
        if (!sp.empty())
        {
            [std::cout](../../io/cout.html) << "old *begin = " << *sp.begin() << '\n';
            *sp.begin() = new_value;
            [std::cout](../../io/cout.html) << "new *begin = " << *sp.begin() << '\n';
        }
    }
     
    int main()
    {
        int array[]{1, 3, 4, 5};
        print(array);
        set_first_element(array, 2);
        print(array);
    }

Output: 
    
    
    array = 1 3 4 5
    old *begin = 1
    new *begin = 2
    array = 2 3 4 5
