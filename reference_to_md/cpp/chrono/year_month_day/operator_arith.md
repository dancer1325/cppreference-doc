Modifies the time point *this represents by the duration dy or dm. 

1) Equivalent to *this = *this + dy;.

2) Equivalent to *this = *this + dm;.

3) Equivalent to *this = *this - dy;.

4) Equivalent to *this = *this - dm;.

For durations that are convertible to both std::chrono::years and std::chrono::months, the `years` overloads (1,3) are preferred if the call would otherwise be ambiguous. 

### [[edit](https://en.cppreference.com/mwiki/index.php?title=cpp/chrono/year_month_day/operator_arith&action=edit&section=1 "Edit section: Example")] Example

Run this code
    
    
    #include <cassert>
    #include <chrono>
    #include <iostream>
     
    int main()
    {
        constexpr auto monthsInYear{12};
        auto ymd{[std::chrono::day](../day.html)(1)/[std::chrono::July](../month.html)/2020};
        [std::cout](../../io/cout.html) << "#1 " << ymd << '\n';
     
        ymd -= [std::chrono::years](../duration.html)(10);
        [std::cout](../../io/cout.html) << "#2 " << ymd << '\n';
        [assert](../../error/assert.html)(ymd.month() == [std::chrono::July](../month.html));
        [assert](../../error/assert.html)(ymd.year() == [std::chrono::year](../year.html)(2010));
     
        ymd += [std::chrono::months](../duration.html)(10 * monthsInYear + 11);
        [std::cout](../../io/cout.html) << "#3 " << ymd << '\n';
        [assert](../../error/assert.html)(ymd.month() == [std::chrono::month](../month.html)(6));
        [assert](../../error/assert.html)(ymd.year() == [std::chrono::year](../year.html)(2021));
     
        // Handling the ymd += months "overflow" case.
        ymd = [std::chrono::May](../month.html)/31/2021; // ok
        [std::cout](../../io/cout.html) << "#4 " << ymd << '\n';
        [assert](../../error/assert.html)(ymd.ok());
     
        ymd += [std::chrono::months](../duration.html){1}; // bad date: June has only 30 days
        [std::cout](../../io/cout.html) << "#5 " << ymd << '\n';
        [assert](../../error/assert.html)(not ymd.ok());
        [assert](../../error/assert.html)(ymd == [std::chrono::June](../month.html)/31/2021);
     
        // Snap to the last day of the month, June 30:
        const auto ymd1 = ymd.year()/ymd.month()/[std::chrono::last](../last_spec.html);
        [std::cout](../../io/cout.html) << "#6 " << ymd1 << '\n';
        [assert](../../error/assert.html)(ymd1.ok());
        [assert](../../error/assert.html)(ymd1 == [std::chrono::June](../month.html)/30/2021);
     
        // Overflow into the next month, July 1 (via converting to/from sys_days):
        const [std::chrono::year_month_day](../year_month_day.html) ymd2 = [std::chrono::sys_days](../system_clock.html){ymd};
        [std::cout](../../io/cout.html) << "#7 " << ymd2 << '\n';
        [assert](../../error/assert.html)(ymd2.ok());
        [assert](../../error/assert.html)(ymd2 == [std::chrono::July](../month.html)/1/2021);
    }

Output: 
    
    
    #1 2020-07-01
    #2 2010-07-01
    #3 2021-06-01
    #4 2021-05-31
    #5 2021-06-31 is not a valid date
    #6 2021/Jun/last
    #7 2021-07-01
