template< class ToDuration, class Clock, class Duration >  
constexpr [std::chrono::time_point](../time_point.html)<Clock, ToDuration>  
round( const [std::chrono::time_point](../time_point.html)<Clock, Duration>& tp ); |  |  (since C++17)  
| |   
  
Returns the nearest time point to tp representable in `ToDuration`, rounding to even in halfway cases. 

The function does not participate in the overload resolution unless `ToDuration` is a specialization of [std::chrono::duration](../duration.html "cpp/chrono/duration") and [std::chrono::treat_as_floating_point_v](../treat_as_floating_point.html)<typename ToDuration::rep> is false.

### [[edit](https://en.cppreference.com/mwiki/index.php?title=cpp/chrono/time_point/round&action=edit&section=3 "Edit section: Possible implementation")] Possible implementation
    
    
    namespace detail
    {
        template<class> inline constexpr bool is_duration_v = false;
        template<class Rep, class Period> inline constexpr bool is_duration_v<
            [std::chrono::duration](../duration.html)<Rep, Period>> = true;
    }
     
    template<class To, class Clock, class FromDuration,
             class = [std::enable_if_t](../../types/enable_if.html)<detail::is_duration_v<To> &&
                    ![std::chrono::treat_as_floating_point_v](../treat_as_floating_point.html)<typename To::rep>>>
    constexpr [std::chrono::time_point](../time_point.html)<Clock, To> round(
        const [std::chrono::time_point](../time_point.html)<Clock, FromDuration>& tp)
    {
        return [std::chrono::time_point](../time_point.html)<Clock, To>{
            [std::chrono::round](../duration/round.html)<To>(tp.time_since_epoch())};
    }

### [[edit](https://en.cppreference.com/mwiki/index.php?title=cpp/chrono/time_point/round&action=edit&section=4 "Edit section: Example")] Example

[[edit]](https://en.cppreference.com/mwiki/index.php?title=Template:cpp/chrono/time_point/round_example&action=edit)

Run this code
    
    
    #include <chrono>
    #include <iostream>
    #include <string>
     
    template<typename TimePoint>
    [std::string](../../string/basic_string.html) to_string(const TimePoint& time_point)
    {
        return [std::to_string](../../string/basic_string/to_string.html)(time_point.time_since_epoch().count());
    }
     
    int main()
    {
        using namespace std::literals::chrono_literals;
        using Sec = [std::chrono::seconds](../duration.html);
     
        [std::cout](../../io/cout.html) << "Time point\t" "Cast\t" "Floor\t" "Round\t" "Ceil\n";
        [std::cout](../../io/cout.html) << "(ms)\t\t"     "(s)\t"  "(s)\t"   "(s)\t"   "(s)\n";
        for (const auto value_ms : {5432ms, 5678ms})
        {
            [std::chrono::time_point](../time_point.html)<[std::chrono::system_clock](../system_clock.html), [std::chrono::milliseconds](../duration.html)>
                time_point_ms(value_ms);
     
            [std::cout](../../io/cout.html)
                << to_string(time_point_ms) << "\t\t"
                << to_string([std::chrono::time_point_cast](time_point_cast.html)<Sec>(time_point_ms)) << '\t'
                << to_string([std::chrono::floor](../duration/floor.html)<Sec>(time_point_ms)) << '\t'
                << to_string([std::chrono::round](../duration/round.html)<Sec>(time_point_ms)) << '\t'
                << to_string([std::chrono::ceil](../duration/ceil.html)<Sec>(time_point_ms)) << '\n';
        }
    }

Output:

    (ms)		(s)	(s)	(s)	(s)
    5432		5	5	5	6
    5678		5	5	6	6
